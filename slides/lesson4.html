import torch
import torch.nn as nn

model = nn.Sequential(
  nn.Linear(784, 64),
  nn.ReLU(),
  nn.Linear(64, 10),
)
optimizer = torch.optim.Adam(model.parameters(), lr=1e-3)
loss_fn = nn.CrossEntropyLoss()
